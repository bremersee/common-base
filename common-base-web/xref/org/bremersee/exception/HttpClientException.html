17  package org.bremersee.exception;
18  
19  import java.util.Collection;
20  import java.util.Collections;
21  import java.util.Map;
22  import lombok.EqualsAndHashCode;
23  import lombok.Getter;
24  import lombok.ToString;
25  import org.bremersee.exception.model.RestApiException;
26  
27  /**
28   * General http client exception.
29   *
30   * @author Christian Bremer
31   */
32  @ToString(callSuper = true)
33  @EqualsAndHashCode(callSuper = false)
34  public class HttpClientException extends RuntimeException
35      implements HttpStatusAware, RestApiExceptionAware, HttpResponseHeadersAware {
36  
37    private final int status;
38  
39    private final Map<String, ? extends Collection<String>> headers;
40  
41    @Getter
42    private final RestApiException restApiException;
43  
44    /**
45     * Instantiates a new http client exception.
46     *
47     * @param status the status
48     * @param message the message
49     * @param headers the headers
50     * @param restApiException the rest api exception
51     */
52    @SuppressWarnings("WeakerAccess")
53    public HttpClientException(
54        final int status,
55        final String message,
56        final Map<String, ? extends Collection<String>> headers,
57        final RestApiException restApiException) {
58  
59      super(message);
60      this.status = status;
61      this.headers = headers != null ? headers : Collections.emptyMap();
62      this.restApiException = restApiException;
63    }
64  
65    @Override
66    public Map<String, ? extends Collection<String>> getMultiValueHeaders() {
67      return headers;
68    }
69  
70    @Override
71    public int status() {
72      return status;
73    }
74  
75  }
