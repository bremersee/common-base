17  package org.bremersee.security.access;
18  
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.Map;
22  import java.util.stream.Collectors;
23  import org.bremersee.common.model.AccessControlEntry;
24  import org.bremersee.common.model.AccessControlList;
25  
26  /**
27   * The acl factory.
28   *
29   * @param <T> the acl type
30   * @author Christian Bremer
31   */
32  public interface AclFactory<T> {
33  
34    /**
35     * Create access control list of the specified type.
36     *
37     * @param owner the owner
38     * @param entries the entries
39     * @return the acl type
40     */
41    T createAccessControlList(String owner, Map<String, ? extends Ace> entries);
42  
43    /**
44     * Gets class of access control list (entity).
45     *
46     * @return the access control list class
47     */
48    default Class<T> getAccessControlListClass() {
49      //noinspection unchecked
50      return (Class<T>) createAccessControlList("someone", Collections.emptyMap()).getClass();
51    }
52  
53    /**
54     * Acl dto factory.
55     *
56     * @return the acl factory
57     */
58    static AclFactory<AccessControlList> dtoFactory() {
59      return new DtoAclFactory();
60    }
61  
62    /**
63     * The dto acl factory.
64     */
65    class DtoAclFactory implements AclFactory<AccessControlList> {
66  
67      @Override
68      public AccessControlList createAccessControlList(
69          final String owner,
70          final Map<String, ? extends Ace> entries) {
71        final AccessControlList acl = new AccessControlList();
72        acl.setOwner(owner);
73        if (entries != null) {
74          acl.setEntries(entries
75              .entrySet()
76              .stream()
77              .map(entry -> {
78                final String permission = entry.getKey();
79                final Ace ace = entry.getValue();
80                return AccessControlEntry
81                    .builder()
82                    .permission(permission)
83                    .guest(ace.isGuest())
84                    .users(new ArrayList<>(ace.getUsers()))
85                    .roles(new ArrayList<>(ace.getRoles()))
86                    .groups(new ArrayList<>(ace.getGroups()))
87                    .build();
88              })
89              .sorted(new AccessControlEntryComparator())
90              .collect(Collectors.toList()));
91        }
92        return acl;
93      }
94    }
95  
96  }
