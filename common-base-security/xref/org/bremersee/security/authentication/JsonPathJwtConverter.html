20  import java.util.Collections;

22  import java.util.Objects;
23  import java.util.Set;
24  import java.util.stream.Collectors;
25  import lombok.EqualsAndHashCode;
26  import lombok.ToString;
27  import org.springframework.core.convert.converter.Converter;
28  import org.springframework.lang.NonNull;
29  import org.springframework.security.core.GrantedAuthority;
30  import org.springframework.security.core.authority.SimpleGrantedAuthority;
31  import org.springframework.security.oauth2.jwt.Jwt;
32  import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationToken;
33  import org.springframework.util.StringUtils;

35  /**
36   * The json path jwt converter.
37   *
38   * @author Christian Bremer
39   */
40  @ToString(doNotUseGetters = true)
41  @EqualsAndHashCode(doNotUseGetters = true)
42  public class JsonPathJwtConverter implements Converter<Jwt, JwtAuthenticationToken> {
43  
44    private String rolesJsonPath = "$.scope";
45  
46    private boolean rolesValueList = false;
47  
48    private String rolesValueSeparator = " ";
49  
50    private String rolePrefix = "SCOPE_";
51  
52    private String nameJsonPath = "$.sub";
53  
54    /**
55     * Sets roles json path.
56     *
57     * @param rolesJsonPath the roles json path
58     */
59    public void setRolesJsonPath(String rolesJsonPath) {
60      if (StringUtils.hasText(rolesJsonPath)) {
61        this.rolesJsonPath = rolesJsonPath;
62      }
63    }
64  
65    /**
66     * Sets roles value list.
67     *
68     * @param rolesValueList the roles value list
69     */
70    public void setRolesValueList(boolean rolesValueList) {
71      this.rolesValueList = rolesValueList;
72    }
73  
74    /**
75     * Sets roles value separator.
76     *
77     * @param rolesValueSeparator the roles value separator
78     */
79    public void setRolesValueSeparator(String rolesValueSeparator) {
80      if (rolesValueSeparator != null) {
81        this.rolesValueSeparator = rolesValueSeparator;
82      }
83    }
84  
85    /**
86     * Sets role prefix.
87     *
88     * @param rolePrefix the role prefix
89     */
90    public void setRolePrefix(String rolePrefix) {
91      if (rolePrefix != null) {
92        this.rolePrefix = rolePrefix;
93      }
94    }
95  
96    /**
97     * Sets name json path.
98     *
99     * @param nameJsonPath the name json path
100    */
101   public void setNameJsonPath(String nameJsonPath) {
102     if (StringUtils.hasText(nameJsonPath)) {
103       this.nameJsonPath = nameJsonPath;
104     }
105   }
106 
107   @Override
108   public JwtAuthenticationToken convert(@NonNull final Jwt source) {
109     final JsonPathJwtParsersonPathJwtParser.html#JsonPathJwtParser">JsonPathJwtParser parser = new JsonPathJwtParser(source);
110     final List<String> roleValues;
111     if (rolesValueList) {
112       final List<?> rawList = parser.read(rolesJsonPath, List.class);
113       roleValues = rawList == null
114           ? Collections.emptyList()
115           : rawList.stream()
116               .filter(Objects::nonNull)
117               .map(Object::toString)
118               .collect(Collectors.toList());
119     } else {
120       final String roleValue = parser.read(rolesJsonPath, String.class);
121       if (StringUtils.hasText(roleValue)) {
122         roleValues = Arrays.asList(roleValue.split(rolesValueSeparator));
123       } else {
124         roleValues = Collections.emptyList();
125       }
126     }
127     final Set<GrantedAuthority> authorities = roleValues.stream()
128         .filter(roleName -> roleName.trim().length() > 0)
129         .map(roleName -> roleName.startsWith(rolePrefix) ? roleName : rolePrefix + roleName)
130         .map(SimpleGrantedAuthority::new)
131         .collect(Collectors.toSet());
132     final String name = parser.read(nameJsonPath, String.class);
133     return new JwtAuthenticationToken(
134         source,
135         authorities,
136         StringUtils.hasText(name) ? name : source.getSubject());
137   }
