67  class MinioRepositoryImplTest {
68  
69    private static MinioRepository repository;
70  
71    private static final MinioOperations minioOperations = mock(MinioOperations.class);
72  
73    private static final String bucket = "testbucket";
74  
75    private static final String name = "test.txt";
76  
77    private static final String etag = "1234";
78  
79    private static final String versionId = "5678";
80  
81    private static final long size = 15L;
82  
83    private static final ZonedDateTime time = ZonedDateTime.now();
84  
85    /**
86     * Sets up.
87     */
88    @BeforeAll
89    static void setUp() {
90      repository = new MinioRepositoryImpl(
91          MinioClient.builder()
92              .endpoint(HttpUrl.get("https://play.min.io"))
93              .credentials("Q3AM3UQ867SPQQA43P2F", "zuf+tfteSlswRu7BJ86wekitnifILbZam1KYY3TG")
94              .build(),
95          null,
96          "testbucket",
97          false,
98          false,
99          Duration.ofDays(1L));
100 
101     when(minioOperations.bucketExists(any(BucketExistsArgs.class))).thenReturn(false);
102 
103     when(minioOperations.isVersioningEnabled(any(IsVersioningEnabledArgs.class))).thenReturn(false);
104 
105     when(minioOperations.putObject(any(PutObjectArgs.class)))
106         .thenReturn(new ObjectWriteResponse(
107             Headers.of(Collections.emptyMap()), bucket, null, name, etag, versionId));
108 
109     when(minioOperations.objectExists(any(StatObjectArgs.class))).thenReturn(true);
110 
111     Map<String, String> headers = new LinkedHashMap<>();
112     headers.put("Content-Type", MediaType.TEXT_PLAIN_VALUE);
113     headers.put("Last-Modified", time.format(Time.HTTP_HEADER_DATE_FORMAT));
114     headers.put("Content-Length", String.valueOf(size));
115     headers.put("ETag", etag);
116     when(minioOperations.statObject(any(StatObjectArgs.class))).thenReturn(new ObjectStat(
117         bucket, name, Headers.of(headers)
118     ));
119 
120     Item item = mock(Item.class);
121     when(item.etag()).thenReturn(etag);
122     when(item.isDeleteMarker()).thenReturn(false);
123     when(item.isDir()).thenReturn(false);
124     when(item.isLatest()).thenReturn(true);
125     when(item.lastModified()).thenReturn(ZonedDateTime.now());
126     when(item.objectName()).thenReturn(name);
127     when(item.size()).thenReturn(size);
128     when(item.versionId()).thenReturn(versionId);
129 
130     when(minioOperations.listObjects(any(ListObjectsArgs.class)))
131         .thenReturn(Collections.singletonList(new Result<>(item)));
132 
133     when(minioOperations.removeObjects(any(RemoveObjectsArgs.class)))
134         .thenReturn(Collections.singletonList(new Result<>(new DeleteError())));
135 
136     when(minioOperations.getPresignedObjectUrl(any(GetPresignedObjectUrlArgs.class)))
137         .thenReturn("https://example.org/somewhere");
138 
139     repository = new MinioRepositoryImpl(
140         minioOperations,
141         null,
142         bucket,
143         true,
144         true,
145         Duration.ofDays(1L));
146   }

180   /**
181    * Save.
182    *
183    * @throws Exception the exception
184    */
185   @Test
186   void save() throws Exception {
187     Optional<ObjectWriteResponse> response = repository.save(
188         MinioObjectId.from("test"),
189         FileAwareMultipartFile.empty(),
190         DeleteMode.NEVER);
191     assertFalse(response.isPresent());
192 
193     MultipartFile multipartFile = new FileAwareMultipartFile(
194         new ByteArrayInputStream("Hello".getBytes(StandardCharsets.UTF_8)),
195         "file",
196         "a.txt",
197         MediaType.TEXT_PLAIN_VALUE);
198     response = repository.save(MinioObjectId.from(name), multipartFile, DeleteMode.ALWAYS);
199     assertTrue(response.isPresent());
200     ObjectWriteResponse obj = response.get();
201     assertEquals(name, obj.object());
202     assertEquals(versionId, obj.versionId());
203     assertEquals(etag, obj.etag());
204     verify(minioOperations).putObject(any(PutObjectArgs.class));
205   }
