31   * @author Christian Bremer

33  @SpringBootTest(
34      classes = {ProxyTestConfiguration.class},
35      webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
36      properties = {"security.basic.enabled=false"})
37  @AutoConfigureWebTestClient
38  @Slf4j
39  class WebClientProxyBuilderTest {
40  
41    @LocalServerPort
42    @SuppressWarnings("unused")
43    private int port;
44  
45    @Autowired
46    private WebTestClient webClient;
47  
48    private String baseUrl() {
49      return "http://localhost:" + port;
50    }
51  
52    private WebClient newWebClient() {
53      return WebClient.builder()
54          .baseUrl(baseUrl())
55          .build();
56    }
57  
58    private ControllerOne newControllerOneClient() {
59      return WebClientProxyBuilder.defaultBuilder()
60          .webClient(newWebClient())
61          .build(ControllerOne.class);
62    }
63  
64    private ControllerTwo newControllerTwoClient() {
65      return WebClientProxyBuilder.defaultBuilder()
66          .webClient(newWebClient())
67          .build(ControllerTwo.class);
68    }
69  
70    private FormDataController newFormDataController() {
71      return WebClientProxyBuilder.defaultBuilder()
72          .webClient(newWebClient())
73          .commonFunctions(InvocationFunctions.builder().build())
74          .build(FormDataController.class);
75    }
76  
77    /**
78     * Call object methods.
79     */
80    @Test
81    void callObjectMethods() {
82      ControllerOne proxy = newControllerOneClient();
83      assertEquals(proxy.hashCode(), proxy.hashCode());
84      assertNotEquals(proxy, null);
85      assertNotNull(proxy.toString());
86    }

88    /**
89     * Call with web test client.
90     */
91    @Test
92    void callWithWebTestClient() {
93      webClient.get().uri("/").exchange().expectStatus().isOk().expectBody(String.class)
94          .isEqualTo(OK_RESPONSE);
95    }
96  
97    /**
98     * Call with web client.
99     */
100   @Test
101   void callWithWebClient() {
102     StepVerifier
103         .create(newWebClient().get().uri(UriBuilder::build).retrieve().bodyToMono(String.class))
104         .assertNext(response -> assertEquals(OK_RESPONSE, response))
105         .expectNextCount(0)
106         .verifyComplete();
107   }
