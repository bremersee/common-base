39  /**
40   * The json path jwt converter test.

44  class JsonPathJwtConverterTest {
45  
46    /**
47     * Convert jwt.
48     */
49    @Test
50    void convert() {
51      Jwt jwt = createJwt();
52      JsonPathJwtConverter converter = new JsonPathJwtConverter();
53      converter.setRolesValueList(true);
54      converter.setRolesJsonPath("$.realm_access.roles");
55      converter.setNameJsonPath("$.preferred_username");
56      converter.setRolePrefix("ROLE_");
57      JwtAuthenticationToken auth = converter.convert(jwt);
58      assertNotNull(auth);
59      assertEquals("anna", auth.getName());
60      assertTrue(auth.getAuthorities().stream()
61          .anyMatch(grantedAuthority -> grantedAuthority.getAuthority()
62              .equalsIgnoreCase("ROLE_LOCAL_USER")));
63  
64      converter.setRolesValueList(false);
65      converter.setRolesJsonPath("$.scope");
66      converter.setNameJsonPath("$.sub");
67      converter.setRolePrefix("SCOPE_");
68      converter.setRolesValueSeparator(" ");
69      auth = converter.convert(jwt);
70      assertNotNull(auth);
71      assertEquals("livia", auth.getName());
72      assertTrue(auth.getAuthorities().stream()
73          .anyMatch(grantedAuthority -> grantedAuthority.getAuthority()
74              .equalsIgnoreCase("SCOPE_EMAIL")));
75      assertTrue(auth.getAuthorities().stream()
76          .anyMatch(grantedAuthority -> grantedAuthority.getAuthority()
77              .equalsIgnoreCase("SCOPE_PROFILE")));
78    }
79  
80    /**
81     * Test convert roles.
82     */
83    @Test
84    void testConvertRoles() {
85  
86      final Map<String, Object> roles = new LinkedHashMap<>();
87      roles.put("roles", Arrays.asList("ADMIN", "ROLE_USER"));
88      final Map<String, Object> claims = new LinkedHashMap<>();
89      claims.put("realm_access", roles);
90  
91      Jwt jwt = mock(Jwt.class);
92      when(jwt.getClaims()).thenReturn(claims);
93  
94      JsonPathJwtConverter converter = new JsonPathJwtConverter();
95      converter.setRolesValueList(true);
96      converter.setRolesJsonPath("$.realm_access.roles");
97      converter.setRolePrefix("ROLE_");
98      final JwtAuthenticationToken authToken = converter.convert(jwt);
99      assertNotNull(authToken);
100     assertNotNull(authToken.getAuthorities());
101     assertTrue(
102         authToken
103             .getAuthorities()
104             .contains(new SimpleGrantedAuthority("ROLE_ADMIN")));
105     assertTrue(
106         authToken
107             .getAuthorities()
108             .contains(new SimpleGrantedAuthority("ROLE_USER")));
109   }

111   private static Jwt createJwt() {
112     Map<String, Object> headers = new HashMap<>();
113     headers.put("alg", "RS256");
114     headers.put("kid", "-V5Nzr9xllRiyURYK-t6pB7C-5E8IKm8-eAPFLVvCt4");
115     headers.put("typ", "JWT");
116 
117     Date exp = new Date(System.currentTimeMillis() + 1000L * 60L * 15L);
118     Date iat = new Date();
119     String sub = UUID.randomUUID().toString();
120     JWTClaimsSet.Builder builder = new Builder();
121     builder.audience("account");
122     builder.expirationTime(exp);
123     builder.issuer("https://openid.dev.bremersee.org/auth/realms/omnia");
124     builder.issueTime(iat);
125     builder.jwtID("48502385-7f37-47da-abc9-a223b8972795");
126     builder.notBeforeTime(iat);
127     builder.subject(sub);
128 
129     builder.claim("sub", "livia");
130     builder.claim("scope", "email profile");
131     builder.claim("email_verified", false);
132     builder.claim("name", "Anna Livia Plurabelle");
133     builder.claim("preferred_username", "anna");
134     builder.claim("given_name", "Anna Livia");
135     builder.claim("family_name", "Plurabelle");
136     builder.claim("email", "anna@example.org");
137 
138     Map<String, Object> roles = new LinkedHashMap<>();
139     roles.put("roles", Collections.singletonList("LOCAL_USER"));
140     builder.claim("realm_access", roles);
141 
142     JWTClaimsSet claimsSet = builder.build();
143     return new Jwt(
144         "an-access-token",
145         iat.toInstant(),
146         exp.toInstant(),
147         headers,
148         claimsSet.getClaims());
149   }
