18    /**
19     * Is guest.
20     */
21    @Test
22    void isGuest() {
23      AceImpl model = new AceImpl();
24      model.setGuest(true);
25      assertTrue(model.isGuest());
26  
27      AceImpl actual = new AceImpl();
28      actual.setGuest(false);
29      assertFalse(actual.isGuest());
30  
31      actual.setGuest(true);
32  
33      assertNotEquals(model, null);
34      assertNotEquals(model, new Object());
35      assertEquals(model, model);
36      assertEquals(model, actual);
37      assertEquals(model.hashCode(), actual.hashCode());
38  
39      assertTrue(model.toString().contains("true"));
40    }
