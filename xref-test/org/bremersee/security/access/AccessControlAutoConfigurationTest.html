48    /**
49     * Acl mapper.
50     */
51    @Test
52    void aclMapper() {
53      AccessControlProperties properties = new AccessControlProperties();
54      AccessControlAutoConfiguration configuration = new AccessControlAutoConfiguration(properties);
55      configuration.init();
56  
57      AclFactory<AclImpl> aclFactory = AclImpl::new;
58      AclMapper<AclImpl> aclMapper = configuration.aclMapper(objectProvider(aclFactory));
59      assertNotNull(aclMapper);
60      assertNotNull(aclMapper.getAclFactory());
61      assertEquals(AclImpl.class, aclMapper.getAclFactory().getAccessControlListClass());
62    }

64    /**
65     * Acl model mapper config adapter.
66     */
67    @Test
68    void aclModelMapperConfigAdapter() {
69      AccessControlProperties properties = new AccessControlProperties();
70      properties.setSwitchAdminAccess(false);
71      properties.setDefaultPermissions(Collections.emptySet());
72      AccessControlAutoConfiguration configuration = new AccessControlAutoConfiguration(properties);
73      configuration.init();
74  
75      AclFactory<AclEntity> aclFactory = AclEntity::new;
76      AclMapper<AclEntity> aclMapper = configuration.aclMapper(objectProvider(aclFactory));
77      ModelMapperConfigurerAdapter adapter = configuration
78          .aclModelMapperConfigAdapter(objectProvider(aclMapper));
79      assertNotNull(adapter);
80  
81      ModelMapper modelMapper = new ModelMapper();
82      adapter.configure(modelMapper);
83  
84      // test from dto to entity
85      AccessControlList dto = AclBuilder.builder()
86          .owner("anna")
87          .guest(true, PermissionConstants.READ)
88          .addUser("otto", PermissionConstants.WRITE)
89          .buildAccessControlList();
90  
91      AclEntity entity = modelMapper.map(dto, AclEntity.class);
92  
93      assertNotNull(entity);
94      assertEquals("anna", entity.getOwner());
95  
96      assertNotNull(entity.get(PermissionConstants.READ));
97      Ace readAce = entity.get(PermissionConstants.READ);
98      assertTrue(readAce.isGuest());
99  
100     Ace writeAce = entity.get(PermissionConstants.WRITE);
101     assertTrue(writeAce.getUsers().contains("otto"));
102 
103     // test from entity to dto
104     AccessControlList actualDto = modelMapper.map(entity, AccessControlList.class);
105     // AccessControlList actualDto = aclMapper.map(entity);
106     assertNotNull(actualDto);
107     assertEquals(dto, actualDto);
108 
109     testHolders(modelMapper);
110   }
111 
112   private void testHolders(ModelMapper modelMapper) {
113     AclHolder aclHolder = new AclHolder();
114     aclHolder.setId("123");
115     aclHolder.setOrder(100);
116     aclHolder.setName("test");
117     aclHolder.setTranslations(Collections.singleton("item"));
118     aclHolder.setAcl(AclBuilder.builder()
119         .owner("anna")
120         .guest(true, PermissionConstants.READ)
121         .addUser("otto", PermissionConstants.WRITE)
122         .buildAccessControlList());
123 
124     AclEntityHolder aclEntityHolder = modelMapper.map(aclHolder, AclEntityHolder.class);
125     assertNotNull(aclEntityHolder.getAcl());
126     assertEquals("anna", aclEntityHolder.getAcl().getOwner());
127 
128     AclHolder newAclHolder = modelMapper.map(aclEntityHolder, AclHolder.class);
129     assertNotNull(newAclHolder.getAcl());
130   }

132   private static <T> ObjectProvider<T> objectProvider(T provides) {
133     //noinspection unchecked
134     ObjectProvider<T> provider = mock(ObjectProvider.class);
135     when(provider.getIfAvailable()).thenReturn(provides);
136     //noinspection unchecked
137     doAnswer((Answer<Void>) invocation -> {
138       invocation.callRealMethod();
139       return null;
140     }).when(provider).ifAvailable(any(Consumer.class));
141     return provider;
142   }
