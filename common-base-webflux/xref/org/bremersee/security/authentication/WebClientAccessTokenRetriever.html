17  package org.bremersee.security.authentication;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import net.minidev.json.JSONObject;
21  import net.minidev.json.JSONValue;
22  import org.bremersee.exception.AccessTokenRetrieverAuthenticationException;
23  import org.bremersee.web.ErrorDetectors;
24  import org.bremersee.web.reactive.function.client.AbstractWebClientErrorDecoder;
25  import org.springframework.http.HttpMethod;
26  import org.springframework.http.MediaType;
27  import org.springframework.security.core.AuthenticationException;
28  import org.springframework.web.reactive.function.BodyInserters;
29  import org.springframework.web.reactive.function.client.ClientResponse;
30  import org.springframework.web.reactive.function.client.WebClient;
31  import reactor.core.publisher.Mono;
32  
33  /**
34   * A reactive implementation of the {@link AccessTokenRetriever}.
35   *
36   * @author Christian Bremer
37   */
38  @Slf4j
39  public class WebClientAccessTokenRetriever
40      extends AbstractWebClientErrorDecoder<AuthenticationException>
41      implements AccessTokenRetriever<Mono<String>> {
42  
43    private final WebClient webClient;
44  
45    private final ReactiveAccessTokenCache accessTokenCache;
46  
47    /**
48     * Instantiates a new access token retriever that uses spring's web client.
49     */
50    public WebClientAccessTokenRetriever() {
51      this(null, null);
52    }
53  
54    /**
55     * Instantiates a new access token retriever that uses spring's web client.
56     *
57     * @param accessTokenCache the access token cache
58     */
59    @SuppressWarnings("unused")
60    public WebClientAccessTokenRetriever(ReactiveAccessTokenCache accessTokenCache) {
61      this(null, accessTokenCache);
62    }
63  
64    /**
65     * Instantiates a new access token retriever that uses spring's web client.
66     *
67     * @param webClient the web client
68     */
69    public WebClientAccessTokenRetriever(
70        WebClient webClient) {
71      this(webClient, null);
72    }
73  
74    /**
75     * Instantiates a new access token retriever that uses spring's web client.
76     *
77     * @param webClient the web client
78     * @param accessTokenCache the access token cache
79     */
80    public WebClientAccessTokenRetriever(
81        WebClient webClient,
82        ReactiveAccessTokenCache accessTokenCache) {
83      this.webClient = webClient != null ? webClient : WebClient.builder().build();
84      this.accessTokenCache = accessTokenCache;
85    }
86  
87    @Override
88    public Mono<String> retrieveAccessToken(final AccessTokenRetrieverProperties properties) {
89      if (log.isDebugEnabled()) {
90        log.debug("Retrieving access token with password flow, properties = {}", properties);
91      }
92      final String cacheKey = properties.createCacheKeyHashed();
93      return Mono.justOrEmpty(accessTokenCache)
94          .flatMap(cache -> cache.findAccessToken(cacheKey))
95          .switchIfEmpty(webClient
96              .method(HttpMethod.POST)
97              .uri(properties.getTokenEndpoint())
98              .contentType(MediaType.APPLICATION_FORM_URLENCODED)
99              .headers(headers -> properties.getBasicAuthProperties()
100                 .ifPresent(basicAuthProperties -> headers.setBasicAuth(
101                     basicAuthProperties.getUsername(),
102                     basicAuthProperties.getPassword())))
103             .body(BodyInserters.fromFormData(properties.createBody()))
104             .retrieve()
105             .onStatus(ErrorDetectors.DEFAULT, this)
106             .bodyToMono(String.class)
107             .map(response -> ((JSONObject) JSONValue.parse(response)).getAsString("access_token"))
108             .flatMap(accessToken -> accessTokenCache != null
109                 ? accessTokenCache.putAccessToken(cacheKey, accessToken)
110                 : Mono.just(accessToken)));
111   }
112 
113   @Override
114   public AuthenticationException buildException(
115       final ClientResponse clientResponse, final String response) {
116     final AccessTokenRetrieverAuthenticationException exception
117         = new AccessTokenRetrieverAuthenticationException(clientResponse.statusCode(), response);
118     log.error("msg=[Retrieving access token with password flow failed.]", exception);
119     return exception;
120   }
121 
122 }
