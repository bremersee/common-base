29  /**
30   * The web client invocation handler.
31   *
32   * @author Christian Bremer
33   */
34  @Slf4j
35  class WebClientInvocationHandler implements InvocationHandler {
36  
37    private final Map<MethodDescription, InvocationFunctions> methodFunctions;
38  
39    private final InvocationFunctions commonFunctions;
40  
41    private final WebClient webClient;
42  
43    private final Class<?> targetClass;
44  
45    /**
46     * Instantiates a new web client invocation handler.
47     *
48     * @param methodFunctions the method functions
49     * @param commonFunctions the common functions
50     * @param webClient the web client
51     * @param targetClass the target class
52     */
53    WebClientInvocationHandler(
54        final Map<MethodDescription, InvocationFunctions> methodFunctions,
55        final InvocationFunctions commonFunctions,
56        final WebClient webClient,
57        final Class<?> targetClass) {
58      this.methodFunctions = methodFunctions;
59      this.commonFunctions = commonFunctions;
60      this.webClient = webClient;
61      this.targetClass = targetClass;
62    }
63  
64    @Override
65    public Object invoke(final Object proxy, final Method method, final Object[] args) {
66      if (ReflectionUtils.isObjectMethod(method)) {
67        if (ReflectionUtils.isEqualsMethod(method)) {
68          return this.equals(args[0]);
69        } else if (ReflectionUtils.isHashCodeMethod(method)) {
70          return this.hashCode();
71        } else if (ReflectionUtils.isToStringMethod(method)) {
72          return "WebClient Proxy of " + (targetClass != null ? targetClass.getName() : "null");
73        } else {
74          return ReflectionUtils.invokeMethod(method, this, args);
75        }
76      }
77      final InvocationParameterslient/proxy/InvocationParameters.html#InvocationParameters">InvocationParameters parameters = new InvocationParameters(targetClass, method, args);
78      final InvocationFunctions functions = InvocationFunctions.merge(
79          commonFunctions,
80          methodFunctions.get(new MethodDescription(method)));
81      final RequestHeadersUriSpec<?> uriSpec = functions.getUriSpecBuilder()
82          .build(parameters, webClient);
83      uriSpec
84          .uri(uriBuilder -> functions.getUriBuilder().build(parameters, uriBuilder))
85          .headers(
86              httpHeaders -> functions.getHeadersBuilder().build(parameters, httpHeaders))
87          .cookies(cookies -> functions.getCookiesBuilder().build(parameters, cookies));
88      if (uriSpec instanceof RequestBodyUriSpec) {
89        functions.getBodyInserter().insert(parameters, (RequestBodyUriSpec) uriSpec);
90      }
91      final ResponseSpec responseSpec = uriSpec.retrieve();
92      responseSpec.onStatus(functions.getErrorDetector(), functions.getErrorDecoder());
93      return functions.getResponseBuilder().build(parameters, responseSpec);
94    }
95  }
