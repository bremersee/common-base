27  /**
28   * The web client proxy builder test.
29   *
30   * @author Christian Bremer
31   */
32  @SpringBootTest(
33      classes = {ProxyTestConfiguration.class},
34      webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT,
35      properties = {"security.basic.enabled=false"})
36  @AutoConfigureWebTestClient
37  @Slf4j
38  class WebClientProxyBuilderTest {
39  
40    @LocalServerPort
41    @SuppressWarnings("unused")
42    private int port;
43  
44    @Autowired
45    private WebTestClient webClient;
46  
47    private String baseUrl() {
48      return "http://localhost:" + port;
49    }
50  
51    private WebClient newWebClient() {
52      return WebClient.builder()
53          .baseUrl(baseUrl())
54          .build();
55    }
56  
57    private ControllerOne newControllerOneClient() {
58      return WebClientProxyBuilder.defaultBuilder()
59          .webClient(newWebClient())
60          .build(ControllerOne.class);
61    }
62  
63    private ControllerTwo newControllerTwoClient() {
64      return WebClientProxyBuilder.defaultBuilder()
65          .webClient(newWebClient())
66          .build(ControllerTwo.class);
67    }
68  
69    private FormDataController newFormDataController() {
70      return WebClientProxyBuilder.defaultBuilder()
71          .webClient(newWebClient())
72          .commonFunctions(InvocationFunctions.builder().build())
73          .build(FormDataController.class);
74    }
75  
76    /**
77     * Call object methods.
78     */
79    @Test
80    void callObjectMethods() {
81      ControllerOne proxy = newControllerOneClient();
82      assertEquals(proxy.hashCode(), proxy.hashCode());
83      assertNotEquals(proxy, null);
84      assertNotNull(proxy.toString());
85    }

87    /**
88     * Call with web test client.
89     */
90    @Test
91    void callWithWebTestClient() {
92      webClient.get().uri("/").exchange().expectStatus().isOk().expectBody(String.class)
93          .isEqualTo(OK_RESPONSE);
94    }
95  
96    /**
97     * Call with web client.
98     */
99    @Test
100   void callWithWebClient() {
101     StepVerifier
102         .create(newWebClient().get().uri(b -> b.build()).retrieve().bodyToMono(String.class))
103         .assertNext(response -> assertEquals(OK_RESPONSE, response))
104         .expectNextCount(0)
105         .verifyComplete();
106   }
