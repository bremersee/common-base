17  package org.bremersee.web.reactive.function.client.proxy.app;
18  
19  import java.util.LinkedHashMap;
20  import java.util.Map;
21  import org.springframework.http.MediaType;
22  import org.springframework.web.bind.annotation.CookieValue;
23  import org.springframework.web.bind.annotation.PostMapping;
24  import org.springframework.web.bind.annotation.RequestHeader;
25  import org.springframework.web.bind.annotation.RequestMapping;
26  import org.springframework.web.bind.annotation.RequestMethod;
27  import org.springframework.web.bind.annotation.RestController;
28  import org.springframework.web.server.ServerWebExchange;
29  import reactor.core.publisher.Mono;
30  
31  /**
32   * The form data controller.
33   *
34   * <p>In a WebFlux application, form data is accessed via {@code ServerWebExchange.getFormData()}.
35   * For this reason the controller can't implement the interface, because it's signature differs from
36   * this implementation.
37   *
38   * @author Christian Bremer
39   */
40  @RestController
41  public class FormDataControllerImpl { // Can't implement FormDataController
42  
43    /**
44     * The constant OK_RESPONSE.
45     */
46    static final String OK_RESPONSE = "OK";
47  
48    /**
49     * Post form data.
50     *
51     * @param exchange the exchange with the form data
52     * @return the result
53     */
54    @PostMapping(path = "/api/oks",
55        produces = MediaType.TEXT_PLAIN_VALUE,
56        consumes = MediaType.APPLICATION_FORM_URLENCODED_VALUE)
57    public Mono<String> addOk(ServerWebExchange exchange) {
58      return exchange.getFormData().map(form -> form.isEmpty() ? "FAILED" : OK_RESPONSE);
59    }
60  
61    /**
62     * Upload mono.
63     *
64     * @param headerValue the header value
65     * @param lastValue the last value
66     * @param exchange the exchange with the form data
67     * @return the result
68     */
69    @RequestMapping(path = "/upload",
70        method = RequestMethod.POST,
71        consumes = MediaType.APPLICATION_FORM_URLENCODED_VALUE,
72        produces = MediaType.APPLICATION_JSON_VALUE)
73    public Mono<Map<String, Object>> upload(
74        @RequestHeader(name = "x-ok-flag") String headerValue,
75        @CookieValue(name = "last") String lastValue,
76        ServerWebExchange exchange) {
77  
78      return exchange.getFormData().map(data -> {
79        LinkedHashMap<String, Object> map = new LinkedHashMap<>();
80        map.put("x-ok-flag", headerValue);
81        map.put("last", lastValue);
82        map.putAll(data);
83        return map;
84      });
85    }
86  }
