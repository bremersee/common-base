39  /**
40   * The jwt authentication token security context factory.
41   *
42   * @author Christian Bremer
43   */
44  @Slf4j
45  public class JwtAuthenticationTokenSecurityContextFactory
46      implements WithSecurityContextFactory<WithJwtAuthenticationToken> {
47  
48    @Override
49    public SecurityContext createSecurityContext(
50        WithJwtAuthenticationToken withJwtAuthenticationToken) {
51  
52      SecurityContext context = SecurityContextHolder.createEmptyContext();
53      Authentication authentication = createJwtConverter(withJwtAuthenticationToken)
54          .convert(createSpringJwt(createJwt(withJwtAuthenticationToken)));
55      context.setAuthentication(authentication);
56      return context;
57    }
58  
59    private Converter<Jwt, ? extends AbstractAuthenticationToken> createJwtConverter(
60        WithJwtAuthenticationToken contextConfig) {
61      JsonPathJwtConverter converter = new JsonPathJwtConverter();
62      converter.setRolesValueSeparator(contextConfig.jwtConverter().rolesValueSeparator());
63      converter.setRolePrefix(contextConfig.jwtConverter().rolePrefix());
64      converter.setNameJsonPath(contextConfig.jwtConverter().nameJsonPath());
65      converter.setRolesJsonPath(contextConfig.jwtConverter().rolesJsonPath());
66      converter.setRolesValueList(contextConfig.jwtConverter().rolesValueList());
67      return converter;
68    }
69  
70    private JWT createJwt(WithJwtAuthenticationToken tokenValues) {
71      JWTClaimsSet.Builder builder = new JWTClaimsSet.Builder()
72          .audience(tokenValues.audience())
73          .expirationTime(new Date(System.currentTimeMillis()
74              + tokenValues.addMillisToExpirationTime()))
75          .issuer(tokenValues.issuer())
76          .issueTime(new Date(System.currentTimeMillis()
77              + tokenValues.addMillisToIssueTime()))
78          .jwtID(tokenValues.jwtId())
79          .notBeforeTime(new Date(System.currentTimeMillis()
80              + tokenValues.addMillisToNotBeforeTime()))
81          .subject(tokenValues.subject());
82      for (Map.Entry<String, Object> entry : createAdditionalClaims(tokenValues).entrySet()) {
83        builder.claim(entry.getKey(), entry.getValue());
84      }
85      JWTClaimsSet claimsSet = builder.build();
86      return new PlainJWT(claimsSet);
87    }
88  
89    private Map<String, Object> createAdditionalClaims(WithJwtAuthenticationToken tokenValues) {
90      Map<String, Object> map = new LinkedHashMap<>();
91      for (Map.Entry<String, Object> entry : createPathMap(tokenValues).entrySet()) {
92        addClaim(entry.getKey(), entry.getValue(), map);
93      }
94      return map;
95    }
96  
97    private void addClaim(String path, Object value, Map<String, Object> map) {
98      int index = path.indexOf('.');
99      if (index < 0) {
100       map.put(path, value);
101     } else {
102       String key = path.substring(0, index);
103       String nextPath = path.substring(index + 1);
104       Object child = map.get(key);
105       if (child instanceof Map) {
106         //noinspection unchecked
107         Map<String, Object> childMap = (Map<String, Object>) child;
108         addClaim(nextPath, value, childMap);
109       } else {
110         Map<String, Object> childMap = new LinkedHashMap<>();
111         map.put(key, childMap);
112         addClaim(nextPath, value, childMap);
113       }
114     }
115   }
116 
117   private Map<String, Object> createPathMap(WithJwtAuthenticationToken tokenValues) {
118     Map<String, Object> map = new TreeMap<>();
119     if (StringUtils.hasText(trimPath(tokenValues.rolesPath()))) {
120       map.put(
121           tokenValues.rolesPath(),
122           Arrays.asList(tokenValues.roles()));
123     }
124     if (StringUtils.hasText(trimPath(tokenValues.scopePath()))) {
125       map.put(
126           tokenValues.scopePath(),
127           Arrays.asList(tokenValues.scope()));
128     }
129     if (StringUtils.hasText(trimPath(tokenValues.namePath()))) {
130       map.put(tokenValues.namePath(), tokenValues.name());
131     }
132     if (StringUtils.hasText(trimPath(tokenValues.preferredUsernamePath()))) {
133       map.put(
134           tokenValues.preferredUsernamePath(),
135           tokenValues.preferredUsername());
136     }
137     if (StringUtils.hasText(trimPath(tokenValues.givenNamePath()))) {
138       map.put(
139           tokenValues.givenNamePath(),
140           tokenValues.givenName());
141     }
142     if (StringUtils.hasText(trimPath(tokenValues.familyNamePath()))) {
143       map.put(
144           tokenValues.familyNamePath(),
145           tokenValues.familyName());
146     }
147     if (StringUtils.hasText(trimPath(tokenValues.emailPath()))) {
148       map.put(tokenValues.emailPath(), tokenValues.email());
149     }
150     return map;
151   }
152 
153   private String trimPath(String path) {
154     if (path == null) {
155       return null;
156     }
157     String tmp = path
158         .replace("..", ".")
159         .trim();
160     while (tmp.startsWith(".")) {
161       tmp = tmp.substring(1).trim();
162     }
163     while (tmp.endsWith(".")) {
164       tmp = tmp.substring(0, tmp.length() - 1).trim();
165     }
166     return tmp;
167   }
168 
169   private Jwt createSpringJwt(JWT jwt) {
170     try {
171       String tokenValue = jwt.serialize();
172       Instant issuedAt = jwt.getJWTClaimsSet().getIssueTime().toInstant();
173       Instant expiresAt = jwt.getJWTClaimsSet().getExpirationTime().toInstant();
174       Map<String, Object> headers = jwt.getHeader().toJSONObject();
175       Map<String, Object> claims = jwt.getJWTClaimsSet().toJSONObject();
176       return new Jwt(tokenValue, issuedAt, expiresAt, headers, claims);
177     } catch (Exception e) {
178       throw new RuntimeException("Creating Spring Jwt failed.", e);
179     }
180   }
