17  package org.bremersee.test.security.authentication;
18  
19  import static org.bremersee.security.core.AuthorityConstants.LOCAL_USER_ROLE_NAME;
20  import static org.bremersee.security.core.AuthorityConstants.USER_ROLE_NAME;
21  import static org.junit.jupiter.api.Assertions.assertEquals;
22  import static org.junit.jupiter.api.Assertions.assertNotNull;
23  import static org.junit.jupiter.api.Assertions.assertTrue;
24  
25  import java.util.List;
26  import org.junit.jupiter.api.Test;
27  import org.springframework.security.core.Authentication;
28  import org.springframework.security.core.authority.SimpleGrantedAuthority;
29  import org.springframework.security.core.context.SecurityContextHolder;
30  import org.springframework.security.oauth2.server.resource.authentication.JwtAuthenticationToken;
31  import org.springframework.test.context.junit.jupiter.SpringJUnitConfig;
32  
33  /**
34   * The jwt authentication token security context factory test.
35   *
36   * @author Christian Bremer
37   */
38  @SpringJUnitConfig
39  class JwtAuthenticationTokenSecurityContextFactoryTest {
40  
41    /**
42     * Create security context.
43     */
44    @WithJwtAuthenticationToken(
45        audience = "account",
46        issuer = "https://openid.dev.bremersee.org/auth/realms/omnia",
47        scope = {"email", "profile"},
48        jwtId = "080836fb-7e74-4a56-92ba-08aeaf9a3852",
49        subject = "1918e152-294b-4701-a2c8-b9090bb5aa06",
50        email = "plurabelle@example.net",
51        roles = {USER_ROLE_NAME, LOCAL_USER_ROLE_NAME})
52    @Test
53    void createSecurityContext() {
54      Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
55      assertNotNull(authentication);
56      assertTrue(authentication instanceof JwtAuthenticationToken);
57      JwtAuthenticationToken authToken = (JwtAuthenticationToken) authentication;
58      assertEquals("account", authToken.getToken().getClaimAsString("aud"));
59      assertEquals(
60          "https://openid.dev.bremersee.org/auth/realms/omnia",
61          authToken.getToken().getClaimAsString("iss"));
62      assertEquals(
63          "080836fb-7e74-4a56-92ba-08aeaf9a3852",
64          authToken.getToken().getClaimAsString("jti"));
65      assertEquals(
66          "1918e152-294b-4701-a2c8-b9090bb5aa06",
67          authToken.getToken().getClaimAsString("sub"));
68      assertEquals(
69          "Anna Livia Plurabelle",
70          authToken.getToken().getClaimAsString("name"));
71      assertEquals(
72          "anna",
73          authToken.getName());
74      assertEquals(
75          "Anna Livia",
76          authToken.getToken().getClaimAsString("given_name"));
77      assertEquals(
78          "Plurabelle",
79          authToken.getToken().getClaimAsString("family_name"));
80      assertEquals(
81          "plurabelle@example.net",
82          authToken.getToken().getClaimAsString("email"));
83      List<String> scopes = authToken.getToken().getClaimAsStringList("scope");
84      assertNotNull(scopes);
85      assertTrue(scopes.contains("email"));
86      assertTrue(scopes.contains("profile"));
87      assertTrue(authToken.getAuthorities().contains(new SimpleGrantedAuthority(USER_ROLE_NAME)));
88      assertTrue(
89          authToken.getAuthorities().contains(new SimpleGrantedAuthority(LOCAL_USER_ROLE_NAME)));
90    }
