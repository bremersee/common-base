116     SearchRequest searchRequest = SearchRequest.builder()

312   /**
313    * Add and modify and delete person.
314    */
315   @Test
316   void addAndModifyAndDeletePerson() {
317     Personapp/Person.html#Person">Person person = new Person();
318     person.setCn("A person");
319     person.setSn("Person");
320     person.setUid("person");
321 
322     String dn = personMapper.mapDn(person);
323     LdapEntry destination = new LdapEntry();
324     personMapper.map(person, destination);
325     destination.setDn(dn);
326     destination.addAttributes(new LdapAttribute(
327         "objectclass",
328         personMapper.getObjectClasses()));
329 
330     StepVerifier.create(ldaptiveTemplate.add(new AddRequest(dn, destination.getAttributes())))
331         .assertNext(result -> assertTrue(result.isSuccess()))
332         .verifyComplete();
333     StepVerifier.create(ldaptiveTemplate.exists(person, personMapper))
334         .assertNext(Assertions::assertTrue)
335         .verifyComplete();
336 
337     person.setSn("Surname");
338     ModifyRequest modifyRequest = personMapper
339         .mapAndComputeModifyRequest(person, destination);
340     StepVerifier.create(ldaptiveTemplate.modify(modifyRequest))
341         .assertNext(result -> assertTrue(result.isSuccess()))
342         .verifyComplete();
343 
344     StepVerifier.create(ldaptiveTemplate.findOne(SearchRequest.objectScopeSearchRequest(dn), personMapper))
345         .assertNext(p -> assertEquals("Surname", p.getSn()))
346         .verifyComplete();
347 
348     person.setSn("");
349     modifyRequest = personMapper.mapAndComputeModifyRequest(person, destination);
350     StepVerifier.create(ldaptiveTemplate.modify(modifyRequest))
351         .assertNext(result -> assertTrue(result.isSuccess()))
352         .verifyComplete();
353 
354     StepVerifier.create(ldaptiveTemplate.findOne(SearchRequest.objectScopeSearchRequest(dn), personMapper))
355         .assertNext(p -> assertNull(p.getSn()))
356         .verifyComplete();
357 
358     StepVerifier.create(ldaptiveTemplate.delete(new DeleteRequest(dn)))
359         .assertNext(result -> assertTrue(result.isSuccess()))
360         .verifyComplete();
361     StepVerifier.create(ldaptiveTemplate.exists(person, personMapper))
362         .assertNext(Assertions::assertFalse)
363         .verifyComplete();
364   }

369   @Test
370   void saveAndRemoveGroup() {
371     Groupe/app/Group.html#Group">Group group = new Group();
372     group.setCn("party");
373     group.setOu("Party Guests");
374 
375     Set<String> members = new LinkedHashSet<>();
376     members.add("uid=anna,ou=people," + baseDn);
377     group.setMembers(members);
378 
379     StepVerifier.create(ldaptiveTemplate.save(group, groupMapper))
380         .assertNext(g -> {
381           assertEquals("party", g.getCn());
382           assertEquals("Party Guests", g.getOu());
383           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
384         })
385         .verifyComplete();
386 
387     SearchRequest searchRequest = SearchRequest.builder()
388         .dn("ou=groups," + baseDn)
389         .filter(FilterTemplate.builder()
390             .filter("(&(objectclass=groupOfUniqueNames)(cn={0}))")
391             .parameters("party")
392             .build())
393         .scope(SearchScope.ONELEVEL)
394         .build();
395 
396     StepVerifier.create(ldaptiveTemplate.findOne(searchRequest, groupMapper))
397         .assertNext(g -> {
398           assertEquals("party", g.getCn());
399           assertEquals("Party Guests", g.getOu());
400           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
401         })
402         .verifyComplete();
403 
404     group.getMembers().add("uid=hans,ou=people," + baseDn);
405     group.getMembers().add("uid=gustav,ou=people," + baseDn);
406     StepVerifier.create(ldaptiveTemplate.save(group, groupMapper))
407         .assertNext(g -> {
408           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
409           assertTrue(g.getMembers().contains("uid=hans,ou=people," + baseDn));
410           assertTrue(g.getMembers().contains("uid=gustav,ou=people," + baseDn));
411         })
412         .verifyComplete();
413 
414     StepVerifier.create(ldaptiveTemplate.findOne(searchRequest, groupMapper))
415         .assertNext(g -> {
416           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
417           assertTrue(g.getMembers().contains("uid=hans,ou=people," + baseDn));
418           assertTrue(g.getMembers().contains("uid=gustav,ou=people," + baseDn));
419         })
420         .verifyComplete();
421 
422     group.getMembers().remove("uid=hans,ou=people," + baseDn);
423     StepVerifier.create(ldaptiveTemplate.save(group, groupMapper))
424         .assertNext(g -> {
425           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
426           assertFalse(g.getMembers().contains("uid=hans,ou=people," + baseDn));
427           assertTrue(g.getMembers().contains("uid=gustav,ou=people," + baseDn));
428         })
429         .verifyComplete();
430 
431     StepVerifier.create(ldaptiveTemplate.findOne(searchRequest, groupMapper))
432         .assertNext(g -> {
433           assertTrue(g.getMembers().contains("uid=anna,ou=people," + baseDn));
434           assertFalse(g.getMembers().contains("uid=hans,ou=people," + baseDn));
435           assertTrue(g.getMembers().contains("uid=gustav,ou=people," + baseDn));
436         })
437         .verifyComplete();
438 
439     StepVerifier.create(ldaptiveTemplate.remove(group, groupMapper))
440         .assertNext(r -> assertTrue(r.isSuccess()))
441         .verifyComplete();
442     StepVerifier.create(ldaptiveTemplate.exists(group, groupMapper))
443         .assertNext(Assertions::assertFalse)
444         .verifyComplete();
445   }
