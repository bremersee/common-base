12  class LdaptiveConnectionConfigFactoryTest {
13  
14    /**
15     * Create connection config.
16     */
17    @Test
18    void createConnectionConfig() {
19      LdaptiveProperties properties = new LdaptiveProperties();
20      properties.setPooled(false);
21      properties.setConnectTimeout(Duration.ofMinutes(1L));
22      properties.setResponseTimeout(Duration.ofMinutes(2L));
23      properties.setUseStartTls(true);
24      properties.setTrustCertificates("trustcert");
25      properties.setAuthenticationCertificate("authcert");
26      properties.setAuthenticationKey("authkey");
27      properties.setBindDn("ou=admin");
28      properties.setBindCredentials("changeit");
29      ConnectionConfig config = LdaptiveConnectionConfigFactory
30          .defaultFactory()
31          .createConnectionConfig(properties);
32      assertNotNull(config);
33      config = LdaptiveConnectionConfigFactory
34          .defaultFactory()
35          .createConnectionConfig(properties, "ou=admin", "changeit");
36      assertNotNull(config);
37    }
