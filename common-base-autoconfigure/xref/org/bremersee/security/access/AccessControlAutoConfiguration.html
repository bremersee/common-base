17  package org.bremersee.security.access;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.bremersee.common.model.AccessControlList;
21  import org.bremersee.converter.ModelMapperConfigurerAdapter;
22  import org.modelmapper.ModelMapper;
23  import org.springframework.beans.factory.ObjectProvider;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
25  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
26  import org.springframework.boot.context.event.ApplicationReadyEvent;
27  import org.springframework.boot.context.properties.EnableConfigurationProperties;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.context.event.EventListener;
31  import org.springframework.core.annotation.Order;
32  import org.springframework.util.Assert;
33  import org.springframework.util.ClassUtils;
34  
35  /**
36   * The access control auto configuration.
37   *
38   * @author Christian Bremer
39   */
40  @EnableConfigurationProperties(AccessControlProperties.class)
41  @ConditionalOnBean(AclFactory.class)
42  @Configuration
43  @Slf4j
44  public class AccessControlAutoConfiguration {
45  
46    /**
47     * The properties.
48     */
49    AccessControlProperties properties;
50  
51    /**
52     * Instantiates a new access control auto configuration.
53     *
54     * @param properties the properties
55     */
56    public AccessControlAutoConfiguration(AccessControlProperties properties) {
57      this.properties = properties;
58    }
59  
60    /**
61     * Init.
62     */
63    @EventListener(ApplicationReadyEvent.class)
64    public void init() {
65      log.info("\n"
66              + "*********************************************************************************\n"
67              + "* {}\n"
68              + "*********************************************************************************\n"
69              + "* defaultPermissions = {}\n"
70              + "* adminRoles = {}\n"
71              + "* switchAdminAccess = {}\n"
72              + "* returnNull = {}\n"
73              + "*********************************************************************************",
74          ClassUtils.getUserClass(getClass()).getSimpleName(),
75          properties.getDefaultPermissions(),
76          properties.getAdminRoles(),
77          properties.isSwitchAdminAccess(),
78          properties.isReturnNull());
79    }
80  
81    /**
82     * Creates acl mapper bean.
83     *
84     * @param <T> the acl entity type
85     * @param aclFactoryProvider the acl factory provider
86     * @return the acl mapper
87     */
88    @Bean
89    public <T extends Acl<? extends Ace>> AclMapper<T> aclMapper(
90        ObjectProvider<AclFactory<T>> aclFactoryProvider) {
91  
92      AclFactory<T> aclFactory = aclFactoryProvider.getIfAvailable();
93      Assert.notNull(aclFactory, "Acl factory must be present.");
94      log.info("Creating bean 'aclMapper' for acl entity '{}' ...",
95          ClassUtils.getUserClass(aclFactory.getAccessControlListClass()));
96      String[] defaultPermissions = properties.getDefaultPermissions().toArray(new String[0]);
97      AclMapperImpl<T> aclMapper = new AclMapperImpl<>(
98          aclFactory,
99          defaultPermissions,
100         properties.isSwitchAdminAccess(),
101         properties.isReturnNull());
102     aclMapper.setAdminRoles(properties.getAdminRoles());
103     return aclMapper;
104   }
105 
106   /**
107    * Creates a model mapper configurer adapter for the acl entity of the acl factory.
108    *
109    * @param <T> the acl entity type
110    * @param aclMapperProvider the acl mapper provider
111    * @return the model mapper configurer adapter
112    */
113   @ConditionalOnClass(ModelMapper.class)
114   @Bean(name = "aclModelMapperConfigurerAdapter")
115   @Order(-1000)
116   public <T extends Acl<? extends Ace>> ModelMapperConfigurerAdapter aclModelMapperConfigAdapter(
117       ObjectProvider<AclMapper<T>> aclMapperProvider) {
118 
119     log.info("Creating bean 'aclModelMapperConfigurerAdapter' ...");
120     return modelMapper -> aclMapperProvider.ifAvailable(aclMapper -> {
121       Class<T> aclEntityClass = aclMapper.getAclFactory()
122           .getAccessControlListClass();
123       log.info("Model mapper for acl is using entity class '{}'.", aclEntityClass.getName());
124       modelMapper
125           .createTypeMap(AccessControlList.class, aclEntityClass)
126           .setConverter(context -> aclMapper.map(context.getSource()));
127       modelMapper
128           .createTypeMap(aclEntityClass, AccessControlList.class)
129           .setConverter(context -> aclMapper.map(context.getSource()));
130     });
131   }
132 
133 }
