101    * @param tokenRetrieverProvider the token retriever provider
102    * @param passwordEncoderProvider the password encoder provider
103    */
104   public ActuatorSecurityAutoConfiguration(
105       SecurityProperties securityProperties,
106       AuthProperties authProperties,
107       ActuatorAuthProperties actuatorAuthProperties,
108       ObjectProvider<JsonPathJwtConverter> jsonPathJwtConverterProvider,
109       ObjectProvider<RestTemplateAccessTokenRetriever> tokenRetrieverProvider,
110       ObjectProvider<PasswordEncoder> passwordEncoderProvider) {
111 
112     this.securityProperties = securityProperties;
113     this.authProperties = authProperties;
114     this.actuatorAuthProperties = actuatorAuthProperties;
115     this.jsonPathJwtConverterProvider = jsonPathJwtConverterProvider;
116     this.tokenRetrieverProvider = tokenRetrieverProvider;
117     this.passwordEncoderProvider = passwordEncoderProvider;
118   }

120   /**
121    * Init.
122    */
123   @EventListener(ApplicationReadyEvent.class)
124   @SuppressWarnings("DuplicatedCode")
125   public void init() {
126     final boolean hasJwkUriSet = StringUtils.hasText(actuatorAuthProperties.getJwkSetUri());
127     log.info("\n"
128             + "*********************************************************************************\n"
129             + "* {}\n"
130             + "*********************************************************************************\n"
131             + "* enable = {}\n"
132             + "* order = {}\n"
133             + "* jwt = {}\n"
134             + "* cors = {}\n"
135             + "*********************************************************************************",
136         ClassUtils.getUserClass(getClass()).getSimpleName(),
137         actuatorAuthProperties.getEnable().name(),
138         actuatorAuthProperties.getOrder(),
139         hasJwkUriSet,
140         actuatorAuthProperties.isEnableCors());
141     if (hasJwkUriSet) {
142       Assert.hasText(actuatorAuthProperties.getPasswordFlow().getTokenEndpoint(),
143           "Token endpoint of actuator password flow must be present.");
144       Assert.hasText(actuatorAuthProperties.getPasswordFlow().getClientId(),
145           "Client ID of actuator password flow must be present.");
146       Assert.notNull(actuatorAuthProperties.getPasswordFlow().getClientSecret(),
147           "Client secret of actuator password flow must be present.");
148     }
149   }

151   @Override
152   public int getOrder() {
153     return actuatorAuthProperties.getOrder();
154   }

156   private EndpointRequest.EndpointRequestMatcher[] unauthenticatedEndpointMatchers() {
157     return actuatorAuthProperties.unauthenticatedEndpointsOrDefaults().stream()
158         .map(EndpointRequest::to)
159         .toArray(EndpointRequest.EndpointRequestMatcher[]::new);
160   }
161 
162   @Override
163   protected void configure(HttpSecurity httpSecurity) throws Exception {
164     log.info("Securing requests to /actuator/**");
165     HttpSecurity http = httpSecurity;
166     ExpressionUrlAuthorizationConfigurer<HttpSecurity>.ExpressionInterceptUrlRegistry reg = http
167         .requestMatcher(EndpointRequest.toAnyEndpoint())
168         .authorizeRequests();
169     if (actuatorAuthProperties.getEnable() == AutoSecurityMode.NONE) {
170       http = reg
171           .anyRequest().permitAll()
172           .and()
173           .httpBasic().disable();
174     } else {
175       if (actuatorAuthProperties.isEnableCors()) {
176         reg = reg.antMatchers(HttpMethod.OPTIONS, "/**").permitAll();
177       }
178       http = reg
179           .requestMatchers(unauthenticatedEndpointMatchers()).permitAll()
180           .requestMatchers(new AndRequestMatcher(
181               EndpointRequest.toAnyEndpoint(),
182               new AntPathRequestMatcher("/**", "GET")))
183           .access(actuatorAuthProperties.buildAccessExpression())
184           .anyRequest()
185           .access(actuatorAuthProperties.buildAdminAccessExpression())
186           .and();
187       if (StringUtils.hasText(actuatorAuthProperties.getJwkSetUri())) {
188         http.authenticationProvider(passwordFlowAuthenticationManager());
189       }
190       http = http
191           .formLogin().disable()
192           .httpBasic().realmName("actuator")
193           .and();
194     }
195     http
196         .csrf().disable()
197         .cors(customizer -> {
198           if (!actuatorAuthProperties.isEnableCors()) {
199             customizer.disable();
200           }
201         })
202         .sessionManagement()
203         .sessionCreationPolicy(SessionCreationPolicy.STATELESS);
204   }

207   @ConditionalOnMissingBean

212         securityProperties,

217   private PasswordFlowAuthenticationManager passwordFlowAuthenticationManager() {
218     return new PasswordFlowAuthenticationManager(
219         actuatorAuthProperties.getPasswordFlow(),
220         jwtDecoder(),
221         jwtConverter(),
222         tokenRetrieverProvider
223             .getIfAvailable(() -> new RestTemplateAccessTokenRetriever(new RestTemplate())));
224   }
225 
226   private JwtDecoder jwtDecoder() {
227     NimbusJwtDecoder nimbusJwtDecoder = NimbusJwtDecoder
228         .withJwkSetUri(actuatorAuthProperties.getJwkSetUri())
229         .jwsAlgorithm(SignatureAlgorithm.from(actuatorAuthProperties.getJwsAlgorithm()))
230         .build();
231     if (StringUtils.hasText(actuatorAuthProperties.getIssuerUri())) {
232       nimbusJwtDecoder.setJwtValidator(
233           JwtValidators.createDefaultWithIssuer(actuatorAuthProperties.getIssuerUri()));
234     }
235     return nimbusJwtDecoder;
236   }
237 
238   private JsonPathJwtConverter jwtConverter() {
239     JsonPathJwtConverter internalJwtConverter = new JsonPathJwtConverter();
240     internalJwtConverter.setNameJsonPath(actuatorAuthProperties.getNameJsonPath());
241     internalJwtConverter.setRolePrefix(actuatorAuthProperties.getRolePrefix());
242     internalJwtConverter.setRolesJsonPath(actuatorAuthProperties.getRolesJsonPath());
243     internalJwtConverter.setRolesValueList(actuatorAuthProperties.isRolesValueList());
244     internalJwtConverter.setRolesValueSeparator(
245         actuatorAuthProperties.getRolesValueSeparator());
246     JsonPathJwtConverter externalJwtConverter = jsonPathJwtConverterProvider.getIfAvailable();
247     JsonPathJwtConverter jwtConverter;
248     if (internalJwtConverter.equals(externalJwtConverter)) {
249       jwtConverter = externalJwtConverter;
250     } else {
251       jwtConverter = internalJwtConverter;
252     }
253     return jwtConverter;
254   }
