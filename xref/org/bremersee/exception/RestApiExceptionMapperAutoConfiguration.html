1   package org.bremersee.exception;
2   
3   import lombok.extern.slf4j.Slf4j;
4   import org.springframework.beans.factory.annotation.Value;
5   import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
6   import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication.Type;
7   import org.springframework.boot.context.event.ApplicationReadyEvent;
8   import org.springframework.boot.context.properties.EnableConfigurationProperties;
9   import org.springframework.context.annotation.Bean;
10  import org.springframework.context.annotation.Configuration;
11  import org.springframework.context.event.EventListener;
12  import org.springframework.util.ClassUtils;
13  
14  /**
15   * The rest api exception mapper auto configuration.
16   *
17   * @author Christian Bremer
18   */
19  @ConditionalOnWebApplication(type = Type.ANY)
20  @Configuration
21  @EnableConfigurationProperties({RestApiExceptionMapperProperties.class})
22  @Slf4j
23  public class RestApiExceptionMapperAutoConfiguration {
24  
25    private final String applicationName;
26  
27    private final RestApiExceptionMapperProperties properties;
28  
29    /**
30     * Instantiates a new rest api exception mapper auto configuration.
31     *
32     * @param applicationName the application name
33     * @param properties the properties
34     */
35    public RestApiExceptionMapperAutoConfiguration(
36        @Value("${spring.application.name:application}") String applicationName,
37        RestApiExceptionMapperProperties properties) {
38      this.applicationName = applicationName;
39      this.properties = properties;
40    }
41  
42    /**
43     * Init.
44     */
45    @EventListener(ApplicationReadyEvent.class)
46    public void init() {
47      log.info("\n"
48              + "*********************************************************************************\n"
49              + "* {}\n"
50              + "*********************************************************************************\n"
51              + "* applicationName = {}\n"
52              + "* apiPaths = {}\n"
53              + "*********************************************************************************",
54          ClassUtils.getUserClass(getClass()).getSimpleName(),
55          applicationName,
56          properties.getApiPaths());
57    }
58  
59    /**
60     * Builds the rest api exception mapper bean.
61     *
62     * @return the rest api exception mapper bean
63     */
64    @Bean
65    public RestApiExceptionMapper restApiExceptionMapper() {
66      return new RestApiExceptionMapperImpl(properties, applicationName);
67    }
68  
69  }
