17  package org.bremersee.context;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  import java.util.Properties;
22  import lombok.extern.slf4j.Slf4j;
23  import org.springframework.boot.autoconfigure.AutoConfigureBefore;
24  import org.springframework.boot.context.event.ApplicationReadyEvent;
25  import org.springframework.boot.context.properties.EnableConfigurationProperties;
26  import org.springframework.context.MessageSource;
27  import org.springframework.context.annotation.Bean;
28  import org.springframework.context.annotation.Configuration;
29  import org.springframework.context.event.EventListener;
30  import org.springframework.context.support.AbstractResourceBasedMessageSource;
31  import org.springframework.context.support.ReloadableResourceBundleMessageSource;
32  import org.springframework.context.support.ResourceBundleMessageSource;
33  import org.springframework.util.ClassUtils;
34  
35  /**
36   * The message source configuration.
37   *
38   * @author Christian Bremer
39   */
40  @AutoConfigureBefore({
41      org.springframework.boot.autoconfigure.context.MessageSourceAutoConfiguration.class
42  })
43  @Configuration
44  @EnableConfigurationProperties({
45      MessageSourceProperties.class
46  })
47  @Slf4j
48  public class MessageSourceAutoConfiguration {
49  
50    private static final String DEFAULT_MESSAGES_BASE_NAME = "messages";
51  
52    private final MessageSourceProperties properties;
53  
54    /**
55     * Instantiates a new message source configuration.
56     *
57     * @param properties the message source properties
58     */
59    public MessageSourceAutoConfiguration(MessageSourceProperties properties) {
60      this.properties = properties;
61    }
62  
63    /**
64     * Init.
65     */
66    @EventListener(ApplicationReadyEvent.class)
67    public void init() {
68      log.info("\n"
69              + "*********************************************************************************\n"
70              + "* {}\n"
71              + "*********************************************************************************\n"
72              + "* baseNames = {}\n"
73              + "* defaultLocale = {}\n"
74              + "* fallbackToSystemLocale = {}\n"
75              + "* defaultEncoding = {}\n"
76              + "*********************************************************************************",
77          ClassUtils.getUserClass(getClass()).getSimpleName(),
78          getBaseNames(),
79          properties.getDefaultLocale(),
80          properties.isFallbackToSystemLocale(),
81          properties.getDefaultEncoding());
82    }
83  
84    /**
85     * Creates message source bean.
86     *
87     * @return the message source bean
88     */
89    @Bean
90    public MessageSource messageSource() {
91      AbstractResourceBasedMessageSource messageSource = properties.isUseReloadableMessageSource()
92          ? createReloadableMessageSource()
93          : createMessageSource();
94      configure(messageSource);
95      return messageSource;
96    }
97  
98    private AbstractResourceBasedMessageSource createReloadableMessageSource() {
99      final ReloadableResourceBundleMessageSource messageSource
100         = new ReloadableResourceBundleMessageSource();
101     if (!properties.getFileEncodings().isEmpty()) {
102       final Properties fileEncodings = new Properties();
103       fileEncodings.putAll(properties.getFileEncodings());
104       messageSource.setFileEncodings(fileEncodings);
105     }
106     return messageSource;
107   }
108 
109   private AbstractResourceBasedMessageSource createMessageSource() {
110     return new ResourceBundleMessageSource();
111   }
112 
113   private void configure(AbstractResourceBasedMessageSource messageSource) {
114     messageSource.setDefaultLocale(properties.defaultLocale());
115     messageSource.setFallbackToSystemLocale(properties.isFallbackToSystemLocale());
116     messageSource.setDefaultEncoding(properties.getDefaultEncoding());
117     final List<String> baseNames = getBaseNames();
118     messageSource.setBasenames(baseNames.toArray(new String[0]));
119     messageSource.setCacheSeconds(properties.getCacheSeconds());
120     messageSource.setAlwaysUseMessageFormat(properties.isAlwaysUseMessageFormat());
121     messageSource.setUseCodeAsDefaultMessage(properties.isUseCodeAsDefaultMessage());
122   }
123 
124   private List<String> getBaseNames() {
125     final List<String> baseNames = new ArrayList<>(properties.getBaseNames());
126     if (baseNames.isEmpty()) {
127       baseNames.add(DEFAULT_MESSAGES_BASE_NAME);
128     }
129     return baseNames;
130   }
131 }
