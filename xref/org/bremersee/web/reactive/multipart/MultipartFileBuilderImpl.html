49  /**
50   * The multipart file builder implementation.
51   *
52   * @author Christian Bremer
53   */
54  @Slf4j
55  public class MultipartFileBuilderImpl implements MultipartFileBuilder {
56  
57    private final File tmpDir;
58  
59    /**
60     * Instantiates a new multipart file builder.
61     */
62    public MultipartFileBuilderImpl() {
63      this((File) null);
64    }
65  
66    /**
67     * Instantiates a new multipart file builder.
68     *
69     * @param tmpDir the tmp dir
70     */
71    public MultipartFileBuilderImpl(String tmpDir) {
72      this(StringUtils.hasText(tmpDir) ? new File(tmpDir) : null);
73    }
74  
75    /**
76     * Instantiates a new multipart file builder.
77     *
78     * @param tmpDir the tmp dir
79     */
80    public MultipartFileBuilderImpl(File tmpDir) {
81      this.tmpDir = tmpDir != null && tmpDir.isDirectory()
82          ? tmpDir
83          : new File(System.getProperty("java.io.tmpdir"));
84    }
85  
86    private Mono<MultipartFile> build(FilePart filePart) {
87      if (filePart == null) {
88        return Mono.just(FileAwareMultipartFile.empty());
89      }
90      File file = newTmpFile();
91      return filePart.transferTo(file)
92          .then(Mono.just(new FileAwareMultipartFile(
93              file,
94              filePart.name(),
95              filePart.filename(),
96              Optional.of(filePart)
97                  .map(part -> part.headers().getContentType())
98                  .map(MimeType::toString)
99                  .orElse(MediaType.APPLICATION_OCTET_STREAM_VALUE))));
100   }
101 
102   private Mono<MultipartFile> build(FormFieldPart formFieldPart) {
103     if (formFieldPart == null || !StringUtils.hasText(formFieldPart.value())) {
104       return Mono.just(FileAwareMultipartFile.empty());
105     }
106     return Mono.just(formFieldPart)
107         .map(part -> {
108           String value = part.value();
109           byte[] data = null;
110           String contentType = null;
111           if (value.toLowerCase().startsWith("data:") && value.indexOf(',') > -1) {
112             DataUrl dataUrl = null;
113             try {
114               dataUrl = new DataUrlSerializer().unserialize(value);
115             } catch (Exception e) {
116               log.debug("Parsing form field as data url failed, "
117                   + "treating value as plain/text (value = {}).", value);
118             }
119             if (dataUrl != null) {
120               data = dataUrl.getData();
121               contentType = dataUrl.getMimeType();
122             }
123           }
124           if (data == null) {
125             data = value.getBytes(StandardCharsets.UTF_8);
126             contentType = MediaType.TEXT_PLAIN_VALUE;
127           }
128           try {
129             return new FileAwareMultipartFile(
130                 new ByteArrayInputStream(data),
131                 tmpDir,
132                 part.name(),
133                 null,
134                 contentType);
135 
136           } catch (IOException e) {
137             throw ServiceException.internalServerError(
138                 "Creating multipart file from form field part failed.", e);
139           }
140         });
141   }
142 
143   @Override
144   public Mono<MultipartFile> build(Part part) {
145     if (part instanceof FilePart) {
146       return build((FilePart) part);
147     }
148     if (part instanceof FormFieldPart) {
149       return build((FormFieldPart) part);
150     }
151     return Mono.just(FileAwareMultipartFile.empty());
152   }
153 
154   @Override
155   public Mono<MultipartFile> build(Flux<? extends Part> parts) {
156     //noinspection unchecked
157     return ((Flux<Part>) parts)
158         .last(new EmptyPart())
159         .flatMap(this::build);
160   }
161 
162   @Override
163   public Mono<List<MultipartFile>> buildList(Flux<? extends Part> parts) {
164     return parts.flatMap(this::build)
165         .collectList();
166   }
167 
168   @Override
169   public Mono<List<MultipartFile>> buildList(
170       MultiValueMap<String, Part> multiPartData, String... requestParameters) {
171 
172     return Flux.fromArray(Optional.ofNullable(requestParameters).orElseGet(() -> new String[0]))
173         .flatMap(requestParameter -> build(multiPartData.getFirst(requestParameter)))
174         .collectList();
175   }
176 
177   @Override
178   public Flux<List<MultipartFile>> buildLists(
179       MultiValueMap<String, Part> multiPartData, String... requestParameters) {
180 
181     return Flux.fromArray(Optional.ofNullable(requestParameters).orElseGet(() -> new String[0]))
182         .flatMap(reqParam -> createList(multiPartData, reqParam));
183   }
184 
185   @SuppressWarnings("unchecked")
186   @Override
187   public Mono<Map<String, MultipartFile>> buildMap(Flux<? extends Part>... parts) {
188     return Flux.fromArray(parts)
189         .flatMap(this::build)
190         .collectMap(MultipartFile::getName, multipartFile -> multipartFile, LinkedHashMap::new);
191   }
192 
193   @Override
194   public Mono<Map<String, MultipartFile>> buildMap(
195       MultiValueMap<String, Part> multiPartData, String... requestParameters) {
196 
197     return Flux.fromArray(Optional.ofNullable(requestParameters).orElseGet(() -> new String[0]))
198         .flatMap(reqParam -> build(multiPartData.getFirst(reqParam))
199             .zipWith(Mono.just(reqParam)))
200         .collectMap(Tuple2::getT2, Tuple2::getT1, LinkedHashMap::new);
201   }
202 
203   @SuppressWarnings("unchecked")
204   @Override
205   public Mono<MultiValueMap<String, MultipartFile>> buildMultiValueMap(
206       Flux<? extends Part>... parts) {
207     return Flux.fromArray(parts)
208         .flatMap(this::buildList)
209         .filter(list -> !list.isEmpty())
210         .collectMap(list -> list.get(0).getName(), list -> list, LinkedHashMap::new)
211         .map(LinkedMultiValueMap::new);
212   }
213 
214   @Override
215   public Mono<MultiValueMap<String, MultipartFile>> buildMultiValueMap(
216       MultiValueMap<String, Part> multiPartData, String... requestParameters) {
217 
218     return Flux.fromArray(Optional.ofNullable(requestParameters).orElseGet(() -> new String[0]))
219         .flatMap(reqParam -> createList(multiPartData, reqParam)
220             .zipWith(Mono.just(reqParam)))
221         .collectMap(Tuple2::getT2, Tuple2::getT1, LinkedHashMap::new)
222         .map(LinkedMultiValueMap::new);
223   }
224 
225   private File newTmpFile() {
226     try {
227       return File.createTempFile("uploaded-", ".tmp", tmpDir);
228     } catch (Exception e) {
229       throw ServiceException.internalServerError("Creating tmp file failed.", e);
230     }
231   }
232 
233   private Mono<List<MultipartFile>> createList(
234       MultiValueMap<String, Part> multiPartData,
235       String requestParameter) {
236 
237     List<Part> contentParts = multiPartData
238         .getOrDefault(requestParameter, Collections.emptyList());
239     return contentParts.isEmpty()
240         ? Mono.empty()
241         : Flux.fromIterable(contentParts).flatMap(this::build).collectList();
242   }
243 
244   private static class EmptyPart implements Part {
245 
246     private EmptyPart() {
247     }
248 
249     @NonNull
250     @Override
251     public String name() {
252       return "";
253     }
254 
255     @NonNull
256     @Override
257     public HttpHeaders headers() {
258       return new HttpHeaders();
259     }
260 
261     @NonNull
262     @Override
263     public Flux<DataBuffer> content() {
264       return Flux.empty();
265     }
266   }
