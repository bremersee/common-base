17  package org.bremersee.security.authentication;
18  
19  import java.time.Duration;
20  import java.util.Date;
21  import java.util.function.Function;
22  import javax.validation.constraints.NotNull;
23  import lombok.Setter;
24  import lombok.extern.slf4j.Slf4j;
25  import org.bremersee.security.authentication.AuthProperties.JwtCache;
26  import org.springframework.data.redis.connection.ReactiveRedisConnectionFactory;
27  import org.springframework.data.redis.core.ReactiveStringRedisTemplate;
28  import org.springframework.util.Assert;
29  import reactor.core.publisher.Mono;
30  
31  /**
32   * The reactive redis access token cache.
33   *
34   * @author Christian Bremer
35   */
36  @Slf4j
37  public class ReactiveRedisAccessTokenCache implements ReactiveAccessTokenCache {
38  
39    private final JwtCache jwtCacheProperties;
40  
41    private final ReactiveStringRedisTemplate redis;
42  
43    @Setter
44    @NotNull
45    private Function<String, Date> findExpirationTimeFn = AccessTokenCache::getExpirationTime;
46  
47    /**
48     * Instantiates a new reactive redis access token cache.
49     *
50     * @param jwtCacheProperties the jwt cache properties
51     * @param connectionFactory the connection factory
52     */
53    public ReactiveRedisAccessTokenCache(
54        JwtCache jwtCacheProperties,
55        ReactiveRedisConnectionFactory connectionFactory) {
56  
57      Assert.notNull(jwtCacheProperties, "Jwt cache properties must be present.");
58      Assert.notNull(connectionFactory, "Redis connection factory must be present.");
59      this.jwtCacheProperties = jwtCacheProperties;
60      this.redis = new ReactiveStringRedisTemplate(connectionFactory);
61    }
62  
63    @Override
64    public Mono<String> findAccessToken(String key) {
65      return redis.opsForValue().get(key)
66          .onErrorResume(
67              throwable -> throwable instanceof RuntimeException,
68              throwable -> {
69                log.error("Getting access token from redis cache failed.", throwable);
70                return Mono.empty();
71              });
72    }
73  
74    @Override
75    public Mono<String> putAccessToken(String key, String accessToken) {
76      Duration duration = jwtCacheProperties.getExpirationTimeThreshold();
77      long millis = System.currentTimeMillis() + duration.toMillis();
78      String dbKey = jwtCacheProperties.addKeyPrefix(key);
79      return Mono.justOrEmpty(findExpirationTimeFn.apply(accessToken))
80          .filter(expirationTime -> expirationTime.getTime() > millis)
81          .flatMap(expirationTime -> redis.opsForValue().set(dbKey, accessToken)
82              .flatMap(success -> success
83                  ? redis.expireAt(dbKey, expirationTime.toInstant().minus(duration))
84                  : Mono.just(false)))
85          .map(result -> accessToken)
86          .onErrorResume(
87              throwable -> throwable instanceof RuntimeException,
88              throwable -> {
89                log.error("Putting access token into redis cache failed.", throwable);
90                return Mono.just(accessToken);
91              })
92          .defaultIfEmpty(accessToken);
93    }
94  
95  }
