17  package org.bremersee.security.authentication;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.bremersee.web.CorsProperties;
21  import org.springframework.beans.factory.ObjectProvider;
22  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
23  import org.springframework.boot.autoconfigure.condition.ConditionalOnExpression;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
25  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
26  import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication.Type;
27  import org.springframework.boot.autoconfigure.security.SecurityProperties;
28  import org.springframework.boot.context.event.ApplicationReadyEvent;
29  import org.springframework.boot.context.properties.EnableConfigurationProperties;
30  import org.springframework.context.annotation.Bean;
31  import org.springframework.context.annotation.Conditional;
32  import org.springframework.context.annotation.Configuration;
33  import org.springframework.context.event.EventListener;
34  import org.springframework.core.env.Environment;
35  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
36  import org.springframework.security.config.annotation.web.configurers.ExpressionUrlAuthorizationConfigurer;
37  import org.springframework.security.core.userdetails.UserDetailsService;
38  import org.springframework.security.crypto.password.PasswordEncoder;
39  
40  /**
41   * The resource server security auto configuration.
42   *
43   * @author Christian Bremer
44   */
45  @ConditionalOnWebApplication(type = Type.SERVLET)
46  @Conditional({ResourceServerAutoSecurityCondition.class})
47  @ConditionalOnClass({
48      HttpSecurity.class,
49      PasswordFlowProperties.class
50  })
51  @ConditionalOnMissingBean(type = {
52      "org.bremersee.actuator.security.authentication.ResourceServerWithActuatorAutoConfiguration"
53  })
54  @EnableConfigurationProperties({
55      CorsProperties.class,
56      SecurityProperties.class,
57      AuthProperties.class})
58  @Configuration
59  @Slf4j
60  public class ResourceServerAutoConfiguration extends AbstractResourceServerAutoConfiguration {
61  
62    /**
63     * Instantiates a new resource server security auto configuration.
64     *
65     * @param environment the environment
66     * @param securityProperties the spring properties
67     * @param authProperties the security properties
68     * @param corsProperties the cors properties
69     * @param jwtConverterProvider the jwt converter provider
70     * @param passwordEncoderProvider the password encoder provider
71     */
72    public ResourceServerAutoConfiguration(
73        Environment environment,
74        SecurityProperties securityProperties,
75        AuthProperties authProperties,
76        CorsProperties corsProperties,
77        ObjectProvider<JsonPathJwtConverter> jwtConverterProvider,
78        ObjectProvider<PasswordEncoder> passwordEncoderProvider) {
79      super(environment, securityProperties, authProperties, corsProperties, jwtConverterProvider,
80          passwordEncoderProvider);
81    }
82  
83    @EventListener(ApplicationReadyEvent.class)
84    @Override
85    public void init() {
86      super.init();
87    }
88  
89    @Override
90    protected ExpressionUrlAuthorizationConfigurer<HttpSecurity>.ExpressionInterceptUrlRegistry init(
91        HttpSecurity httpSecurity) throws Exception {
92      return httpSecurity.authorizeRequests();
93    }
94  
95    @ConditionalOnExpression("'${spring.security.oauth2.resourceserver.jwt.jwk-set-uri:}'.empty")
96    @ConditionalOnMissingBean
97    @Bean
98    @Override
99    public UserDetailsService userDetailsServiceBean() {
100     return new InMemoryUserDetailsAutoConfiguration().inMemoryUserDetailsManager(
101         getSecurityProperties(),
102         getAuthProperties(),
103         getPasswordEncoderProvider());
104   }
105 
106 }
