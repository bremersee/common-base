19  import java.util.Arrays;
20  import java.util.Collection;
21  import java.util.Collections;
22  import java.util.Map;
23  import org.bremersee.common.model.AccessControlList;

26  /**
27   * The access controller.
28   *
29   * @author Christian Bremer
30   */
31  public interface AccessController {
32  
33    /**
34     * Creates an access controller from the given access control list.
35     *
36     * @param acl the access control list
37     * @return the access controller
38     */
39    static AccessController from(@Nullable AccessControlList acl) {
40      return acl == null ? new Impl() : new Impl(AclBuilder.builder().from(acl).buildAcl());
41    }
42  
43    /**
44     * Creates an access controller from the given access control list.
45     *
46     * @param acl the access control list
47     * @return the access controller
48     */
49    @SuppressWarnings("rawtypes")
50    static AccessController from(@Nullable Acl acl) {
51      return acl == null ? new Impl() : new Impl(AclBuilder.builder().from(acl).buildAcl());
52    }
53  
54    /**
55     * Determines whether the given user with the given roles and groups has the specified
56     * permission.
57     *
58     * @param user       the user
59     * @param roles      the roles
60     * @param groups     the groups
61     * @param permission the permission
62     * @return {@code true} if the user has the permission, otherwise {@code false}
63     */
64    boolean hasPermission(
65        @Nullable String user,
66        @Nullable Collection<String> roles,
67        @Nullable Collection<String> groups,
68        @Nullable String permission);
69  
70    /**
71     * Determines whether the given user with the given roles and groups has at least one of the
72     * specified permissions.
73     *
74     * @param user        the user
75     * @param roles       the roles
76     * @param groups      the groups
77     * @param permissions the permissions
78     * @return {@code true} if the user has at least one permission, otherwise {@code false}
79     */
80    default boolean hasAnyPermission(
81        @Nullable String user,
82        @Nullable Collection<String> roles,
83        @Nullable Collection<String> groups,
84        @Nullable String... permissions) {
85  
86      return hasAnyPermission(
87          user,
88          roles,
89          groups,
90          permissions == null ? Collections.emptyList() : Arrays.asList(permissions));
91    }
92  
93    /**
94     * Determines whether the given user with the given roles and groups has at least one of the
95     * specified permissions.
96     *
97     * @param user        the user
98     * @param roles       the roles
99     * @param groups      the groups
100    * @param permissions the permissions
101    * @return {@code true} if the user has at least one permission, otherwise {@code false}
102    */
103   default boolean hasAnyPermission(
104       @Nullable String user,
105       @Nullable Collection<String> roles,
106       @Nullable Collection<String> groups,
107       @Nullable Collection<String> permissions) {
108 
109     return permissions != null
110         && !permissions.isEmpty()
111         && permissions
112         .stream()
113         .anyMatch(permission -> hasPermission(user, roles, groups, permission));
114   }
115 
116   /**
117    * Determines whether the given user with the given roles and groups has all specified
118    * permissions.
119    *
120    * @param user        the user
121    * @param roles       the roles
122    * @param groups      the groups
123    * @param permissions the permissions
124    * @return {@code true} if the user has all permissions, otherwise {@code false}
125    */
126   default boolean hasAllPermissions(
127       String user,
128       Collection<String> roles,
129       Collection<String> groups,
130       String... permissions) {
131 
132     return hasAllPermissions(
133         user,
134         roles,
135         groups,
136         permissions == null ? Collections.emptyList() : Arrays.asList(permissions));
137   }
138 
139   /**
140    * Determines whether the given user with the given roles and groups has all specified
141    * permissions.
142    *
143    * @param user        the user
144    * @param roles       the roles
145    * @param groups      the groups
146    * @param permissions the permissions
147    * @return {@code true} if the user has all permissions, otherwise {@code false}
148    */
149   default boolean hasAllPermissions(
150       String user,
151       Collection<String> roles,
152       Collection<String> groups,
153       Collection<String> permissions) {
154 
155     return permissions != null
156         && !permissions.isEmpty()
157         && permissions
158         .stream()
159         .allMatch(permission -> hasPermission(user, roles, groups, permission));
160   }
161 
162   /**
163    * The default access controller implementation.
164    */
165   class Impl implements AccessController {
166 
167     private final Acl<? extends Ace> acl;
168 
169     /**
170      * Instantiates a new access controller.
171      */
172     Impl() {
173       acl = null;
174     }
175 
176     /**
177      * Instantiates a new access controller.
178      *
179      * @param acl the acl
180      */
181     Impl(final Acl<? extends Ace> acl) {
182       this.acl = acl;
183     }
184 
185     @Override
186     public boolean hasPermission(
187         final String user,
188         final Collection<String> roles,
189         final Collection<String> groups,
190         final String permission) {
191 
192       if (acl == null) {
193         return false;
194       }
195       if (permission == null) {
196         return false;
197       }
198       if (user != null && user.equals(acl.getOwner())) {
199         return true;
200       }
201       final Map<String, ? extends Ace> map = acl.entryMap();
202       if (map == null) {
203         return false;
204       }
205       final Ace ace = map.get(permission.toLowerCase());
206       if (ace == null) {
207         return false;
208       }
209       if (ace.isGuest()) {
210         return true;
211       }
212       if (ace.getUsers().contains(user)) {
213         return true;
214       }
215       if (roles != null && roles.stream().anyMatch(role -> ace.getRoles().contains(role))) {
216         return true;
217       }
218       return groups != null && groups.stream().anyMatch(group -> ace.getGroups().contains(group));
219     }
220   }
